type FlagConfig struct {
    Name     string
    Default  string
    EnvVar   string
    IsSecret bool  // Forces post-parse fallback
}

func addAuthFlag(cmd *cobra.Command, flags []FlagConfig) {
    for _, f := range flags {
        var default_ string
        if !f.IsSecret {
            default_ = envOr(f.EnvVar, f.Default)
        }
        cmd.Flags().StringVar(/* dst */, f.Name, default_, "...")

**Option B: Global PreRunE hook for all string flags (Lower cost, higher automation)**
```go
func chainSecretEnvFallbacks(cmd *cobra.Command, secretFlags map[string]string) {
    // secretFlags: flag-name -&gt; env-var
    chainPreRunE(cmd, func(cmd *cobra.Command, _ []string) error {
        for flagName, envVar := range secretFlags {
            if !cmd.Flags().Changed(flagName) {
                if v := os.Getenv(envVar); v != "" {
                    cmd.Flags().Set(flagName, v)
                }
            }
        }
        return nil
    })
}
```

**Benefit:** Single map declares all secrets upfront, scales to any number of flags.
